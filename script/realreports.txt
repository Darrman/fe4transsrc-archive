# This file contains the scripts which play when something
# notable happens at the end of a 'real' battle.

# (weapon, ring, etc) Obtained
# 0x190187
# @@1436422@@
[$00$27][sp]obtained![n]

[exit]

# (gold) Obtained
# 0x190198
# @@1436471@@
[$00$25][sp]gold obtained![n]

[exit]

# 0x1901ab
# @@1436520@@

[exit]

# 0x1901bb
